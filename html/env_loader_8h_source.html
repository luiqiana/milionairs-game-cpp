52inline void loadEnv() {

55 if(!file.is_open()) {
56 // Check Working Directories in case of using CLion or other IDEs
57 std::cerr << "[ENV] Error: .env file not found at ./database/.env!" << std::endl;

61 std::string line;
62 while(std::getline(file, line)) {
63 // Skip empty lines and comments
64 if(line.empty() || line[0] == '#') continue;
65
66 size_t pos = line.find('=');
67 if(pos != std::string::npos) {
68 std::string key = trim(line.substr(0, pos));
